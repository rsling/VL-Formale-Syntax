\section{Schemata und Prinzipien}

\subsection{Kopf-Argument-Schema}

\textit{hd-arg-phrase} $\Rightarrow$
\begin{avm}
  \[
    subcat & \@1 \\
    hd-dtr|subcat & \@1 $\oplus$ \< \@2 \> \\
    nhd-dtr & \@2
  \]
\end{avm}

\subsection{Kopfmerkmalprinzip}

\noindent \textit{headed-phrase} $\Rightarrow$
\begin{avm}
  \[
    head & \@1 \\
    hd-dtr|head & \@1
  \]
\end{avm}
